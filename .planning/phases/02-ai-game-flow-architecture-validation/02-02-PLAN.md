**Step 2: Create `createAIController(difficulty)` factory function**

```javascript
function createAIController(difficulty) {
  const cfg = AI_DIFFICULTIES[difficulty];
  return {
    state: AI_STATE.APPROACH,
    decisionCooldown: 0,
    attackSequenceFrames: 0,  // counts frames in current attack sequence
    attackHoldFrames: 0,      // for power attack simulation
    comboHitsRemaining: 0,    // fast attacks remaining in current combo
    isHolding: false,         // true when simulating held attack for power
    retreatFrames: 0,         // countdown for back-off duration

    generateInput(cpu, player, frameCount) {
      const input = {
        left: false, right: false, up: false, down: false,
        attack: false, block: false, ultimate: false, enter: false,
        attackPressed: false, attackReleased: false, attackBuffered: false
      };

      // Don't act when knocked down
      if (cpu.knockedDown || cpu.state === 'knockdown') return input;

      const dist = Math.abs(cpu.x - player.x);
      const facingPlayer = (player.x > cpu.x) ? 1 : -1;
      const playerAttacking = (player.state === 'fastAttack' || player.state === 'powerAttack');
      const playerInRecovery = (player.attackPhase === 'recovery');
      const cpuHealthPct = cpu.health / cpu.def.stats.health;

      // === ATTACK SEQUENCE HANDLING ===
      // If in an attack sequence (combo or power), handle frame-by-frame
      if (this.comboHitsRemaining > 0) {
        this.attackSequenceFrames++;
        // Pattern: 2 frames attack=true, 1 frame attack=false (triggers fast attack)
        const cyclePos = this.attackSequenceFrames % 3;
        if (cyclePos < 2) {
          input.attack = true;
          input.attackBuffered = true;
        } else {
          input.attack = false;
          this.comboHitsRemaining--;
        }
        // Edge detection for fighter charge system
        input.attackPressed = (input.attack && cyclePos === 0);
        input.attackReleased = (!input.attack && cyclePos === 2);
        return input;
      }
      if (this.isHolding) {
        this.attackHoldFrames++;
        if (this.attackHoldFrames < 14) {
          input.attack = true;  // Hold for 14 frames (> 12 threshold)
        } else {
          input.attack = false;  // Release triggers power attack
          input.attackReleased = true;
          this.isHolding = false;
          this.attackHoldFrames = 0;
        }
        return input;
      }

      // === DECISION COOLDOWN (reaction delay) ===
      this.decisionCooldown = Math.max(0, this.decisionCooldown - 1);
      if (this.decisionCooldown > 0) {
        // Between decisions: maintain movement based on current state
        if (this.state === AI_STATE.APPROACH) {
          if (facingPlayer === 1) input.right = true; else input.left = true;
        } else if (this.state === AI_STATE.BACK_OFF) {
          if (facingPlayer === 1) input.left = true; else input.right = true;
        }
        return input;
      }

      // === STATE TRANSITIONS ===
      this.decisionCooldown = cfg.reactionFrames;

      // Back-off: low health and close
      if (cpuHealthPct < cfg.retreatHealth && dist < cfg.approachDist) {
        this.state = AI_STATE.BACK_OFF;
        this.retreatFrames = 30 + Math.floor(Math.random() * 30);
      }
      // Punish: player in recovery and close enough
      else if (playerInRecovery && dist < cfg.pressureDist + 20) {
        this.state = AI_STATE.PUNISH;
      }
      // Pressure: close enough to attack
      else if (dist < cfg.pressureDist) {
        this.state = AI_STATE.PRESSURE;
      }
      // Approach: too far
      else {
        this.state = AI_STATE.APPROACH;
      }

      // === BLOCKING (reactive, independent of state) ===
      if (playerAttacking && dist < 80 && Math.random() < cfg.blockChance) {
        input.block = true;
        return input;
      }

      // === BEHAVIOR BY STATE ===
      switch (this.state) {
        case AI_STATE.APPROACH:
          if (facingPlayer === 1) input.right = true; else input.left = true;
          // Occasional jump
          if (Math.random() < cfg.jumpChance) input.up = true;
          break;

        case AI_STATE.PRESSURE:
          // Decide: combo, power attack, or single attack
          if (Math.random() < cfg.powerAttackChance) {
            // Start power attack hold
            this.isHolding = true;
            this.attackHoldFrames = 0;
            input.attack = true;
            input.attackPressed = true;
          } else if (Math.random() < cfg.comboChance) {
            // Start combo chain (2-4 hits based on difficulty)
            const maxHits = cfg === AI_DIFFICULTIES[2] ? 4 : (cfg === AI_DIFFICULTIES[1] ? 3 : 2);
            this.comboHitsRemaining = 1 + Math.floor(Math.random() * maxHits);
            this.attackSequenceFrames = 0;
            input.attack = true;
            input.attackPressed = true;
            input.attackBuffered = true;
          } else {
            // Single fast attack
            input.attack = true;
            input.attackPressed = true;
            input.attackBuffered = true;
          }
          break;

        case AI_STATE.BACK_OFF:
          if (facingPlayer === 1) input.left = true; else input.right = true;
          this.retreatFrames--;
          if (this.retreatFrames <= 0) this.state = AI_STATE.APPROACH;
          // Block while retreating if player is attacking
          if (playerAttacking) input.block = true;
          break;

        case AI_STATE.PUNISH:
          // Rush in and attack during player recovery
          if (dist > cfg.pressureDist) {
            if (facingPlayer === 1) input.right = true; else input.left = true;
          } else {
            if (Math.random() < 0.5 && cfg.comboChance > 0.2) {
              this.comboHitsRemaining = 2;
              this.attackSequenceFrames = 0;
              input.attack = true;
              input.attackPressed = true;
              input.attackBuffered = true;
            } else {
              input.attack = true;
              input.attackPressed = true;
              input.attackBuffered = true;
            }
          }
          break;
      }

      return input;
    }
  };
}
```
